CREATE TABLE events(
 id INTEGER PRIMARY KEY AUTOINCREMENT,
 title TEXT NOT NULL,
 type TEXT NOT NULL,
 start TEXT NOT NULL,
 end TEXT NOT NULL,
 created TEXT NOT NULL
)

UPDATE events
 SET start = start || " 10:00",
  [end] = [end] || " 10:00",
  created = created || " 14:00"
WHERE
 type = 'Pikett';

UPDATE events
 SET start = start || " 01:00",
  [end] = [end] || " 23:00",
  created = created || " 14:00"
WHERE
 type NOT LIKE 'Pikett';
